def combine(z,name):
    d[name]=pd.DataFrame()
    print('creating data_frame......')
    for i in z:
        df=pd.read_csv(i)
        d[name]=d[name].append(df,ignore_index=True)
    print('done')

def pie_plot(data,labels):
    fig,ax=plt.subplots()
    patches,texts,pcts=ax.pie(data,labels=labels,autopct='%.1f%%')
    plt.setp(texts,color='white')
    plt.show()
